51 template <typename T, typename Handler>
52 void async_write(const T& t, Handler handler)
53 {
54 // Serialize the data first so we know how large it is.
55 std::ostringstream archive_stream;
56 boost::archive::text_oarchive archive(archive_stream);
57 archive << t;
58 outbound_data_ = archive_stream.str();
59
60 // Format the header.
61 std::ostringstream header_stream;
62 header_stream << std::setw(header_length)
63 << std::hex << outbound_data_.size();
64 if (!header_stream || header_stream.str().size() != header_length)
65 {
66 // Something went wrong, inform the caller.
67 std::error_code error(asio::error::invalid_argument);
68 asio::post(socket_.get_executor(), std::bind(handler, error));
69 return;
70 }
71 outbound_header_ = header_stream.str();
72
73 // Write the serialized data to the socket. We use "gather-write" to send
74 // both the header and the data in a single write operation.
75 std::vector<asio::const_buffer> buffers;
76 buffers.push_back(asio::buffer(outbound_header_));
77 buffers.push_back(asio::buffer(outbound_data_));
78 asio::async_write(socket_, buffers, handler);
79 }

95 template <typename T, typename Handler>
96 void handle_read_header(const std::error_code& e,
97 T& t, std::tuple<Handler> handler)
98 {
99 if (e)
100 {
101 std::get<0>(handler)(e);
102 }
103 else
104 {
105 // Determine the length of the serialized data.
106 std::istringstream is(std::string(inbound_header_, header_length));
107 std::size_t inbound_data_size = 0;
108 if (!(is >> std::hex >> inbound_data_size))
109 {
110 // Header doesn't seem to be valid. Inform the caller.
111 std::error_code error(asio::error::invalid_argument);
112 std::get<0>(handler)(error);
113 return;
114 }
115
116 // Start an asynchronous call to receive the data.
117 inbound_data_.resize(inbound_data_size);
118 void (connection::*f)(
119 const std::error_code&,
120 T&, std::tuple<Handler>)
121 = &connection::handle_read_data<T, Handler>;
122 asio::async_read(socket_, asio::buffer(inbound_data_),
123 std::bind(f, this,
124 asio::placeholders::error, std::ref(t), handler));
125 }
126 }

129 template <typename T, typename Handler>
130 void handle_read_data(const std::error_code& e,
131 T& t, std::tuple<Handler> handler)
132 {
133 if (e)
134 {
135 std::get<0>(handler)(e);
136 }
137 else
138 {
139 // Extract the data structure from the data just received.
140 try
141 {
142 std::string archive_data(&inbound_data_[0], inbound_data_.size());
143 std::istringstream archive_stream(archive_data);
144 boost::archive::text_iarchive archive(archive_stream);
145 archive >> t;
146 }
147 catch (std::exception& e)
148 {
149 // Unable to decode data.
150 std::error_code error(asio::error::invalid_argument);
151 std::get<0>(handler)(error);
152 return;
153 }
154
155 // Inform caller that data has been received ok.
156 std::get<0>(handler)(e);
157 }
158 }
