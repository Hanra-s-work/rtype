35template <typename Impl, typename Work, typename Handler>
36class composed_op<Impl, Work, Handler>
37 : public base_from_cancellation_state<Handler>
38{
39public:
40 template <typename I, typename W, typename H>
41 composed_op(I&& impl,
42 W&& work,
43 H&& handler)
44 : base_from_cancellation_state<Handler>(
45 handler, enable_terminal_cancellation()),
46 impl_(static_cast<I&&>(impl)),
47 work_(static_cast<W&&>(work)),
48 handler_(static_cast<H&&>(handler)),
49 invocations_(0)
50 {
51 }
52
53 composed_op(composed_op&& other)
54 : base_from_cancellation_state<Handler>(
55 static_cast<base_from_cancellation_state<Handler>&&>(other)),
56 impl_(static_cast<Impl&&>(other.impl_)),
57 work_(static_cast<Work&&>(other.work_)),
58 handler_(static_cast<Handler&&>(other.handler_)),
59 invocations_(other.invocations_)
60 {
61 }

63 typedef typename composed_work_guard<
64 typename Work::head_type>::executor_type io_executor_type;

66 io_executor_type get_io_executor() const noexcept

85 template <typename... T>
86 void operator()(T&&... t)
87 {
88 if (invocations_ < ~0u)
89 ++invocations_;
90 this->get_cancellation_state().slot().clear();
91 impl_(*this, static_cast<T&&>(t)...);
92 }

94 template <typename... Args>
95 auto complete(Args&&... args)

193template <typename Implementation, typename Executors, typename... Signatures>
194class initiate_composed
195{
196public:
197 typedef typename composed_io_executors<Executors>::head_type executor_type;
198
199 template <typename I>
200 initiate_composed(I&& impl, composed_io_executors<Executors>&& executors)
201 : implementation_(std::forward<I>(impl)),
202 executors_(std::move(executors))
203 {
204 }
205
206 executor_type get_executor() const noexcept
207 {
208 return executors_.head_;
209 }
210
211 template <typename Handler, typename... Args>
212 void operator()(Handler&& handler, Args&&... args) const &
213 {
214 composed_op<decay_t<Implementation>, composed_work<Executors>,
215 decay_t<Handler>, Signatures...>(implementation_,
216 composed_work<Executors>(executors_),
217 static_cast<Handler&&>(handler))(static_cast<Args&&>(args)...);
218 }
219
220 template <typename Handler, typename... Args>
221 void operator()(Handler&& handler, Args&&... args) &&
222 {
223 composed_op<decay_t<Implementation>, composed_work<Executors>,
224 decay_t<Handler>, Signatures...>(
225 static_cast<Implementation&&>(implementation_),
226 composed_work<Executors>(executors_),
227 static_cast<Handler&&>(handler))(static_cast<Args&&>(args)...);
228 }
229
230private:
231 Implementation implementation_;
232 composed_io_executors<Executors> executors_;
233};
