228template <typename Protocol, typename Executor,
229 typename Iterator, typename ConnectCondition>
230Iterator connect(basic_socket<Protocol, Executor>& s, Iterator begin,
231 Iterator end, ConnectCondition connect_condition,
232 asio::error_code& ec,
233 constraint_t<
234 is_connect_condition<ConnectCondition, Iterator>::value
235 >)
236{
237 ec = asio::error_code();
238
239 for (Iterator iter = begin; iter != end; ++iter)
240 {
241 iter = (detail::call_connect_condition(connect_condition, ec, iter, end));
242 if (iter != end)
243 {
244 s.close(ec);
245 s.connect(*iter, ec);
246 if (!ec)
247 return iter;
248 }
249 else
250 break;
251 }
252
253 if (!ec)
254 ec = asio::error::not_found;
255
256 return end;
257}

352 //private:
353 template <typename Iterator>
354 void process(asio::error_code ec,
355 int start, Iterator begin, Iterator end)
356 {
357 Iterator iter = begin;
358 std::advance(iter, index_);
359
360 switch (start_ = start)
361 {
362 case 1:
363 for (;;)
364 {
365 this->check_condition(ec, iter, end);
366 index_ = std::distance(begin, iter);
367
368 if (iter != end)
369 {
370 socket_.close(ec);
371 ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_connect"));
372 socket_.async_connect(*iter,
373 static_cast<range_connect_op&&>(*this));
374 return;
375 }
376
377 if (start)
378 {
379 ec = asio::error::not_found;
380 ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_connect"));
381 asio::post(socket_.get_executor(),
382 detail::bind_handler(
383 static_cast<range_connect_op&&>(*this), ec));
384 return;
385 }
386
387 /* fall-through */ default:
388
389 if (iter == end)
390 break;
391
392 if (!socket_.is_open())
393 {
394 ec = asio::error::operation_aborted;
395 break;
396 }
397
398 if (!ec)
399 break;
400
401 if (this->cancelled() != cancellation_type::none)
402 {
403 ec = asio::error::operation_aborted;
404 break;
405 }
406
407 ++iter;
408 ++index_;
409 }
410
411 static_cast<RangeConnectHandler&&>(handler_)(
412 static_cast<const asio::error_code&>(ec),
413 static_cast<const typename Protocol::endpoint&>(
414 ec || iter == end ? typename Protocol::endpoint() : *iter));
415 }
416 }
417
418 basic_socket<Protocol, Executor>& socket_;
419 EndpointSequence endpoints_;
420 std::size_t index_;
421 int start_;
422 RangeConnectHandler handler_;
