32// Class to adapt a redirect_error_t as a completion handler.
33template <typename Handler>
34class redirect_error_handler
35{
36public:
37 typedef void result_type;
38
39 template <typename CompletionToken>
40 redirect_error_handler(redirect_error_t<CompletionToken> e)
41 : ec_(e.ec_),
42 handler_(static_cast<CompletionToken&&>(e.token_))
43 {
44 }
45
46 template <typename RedirectedHandler>
47 redirect_error_handler(asio::error_code& ec,
48 RedirectedHandler&& h)
49 : ec_(ec),
50 handler_(static_cast<RedirectedHandler&&>(h))
51 {
52 }
53
54 void operator()()
55 {
56 static_cast<Handler&&>(handler_)();
57 }
58
59 template <typename Arg, typename... Args>
60 enable_if_t<
61 !is_same<decay_t<Arg>, asio::error_code>::value
62 >
63 operator()(Arg&& arg, Args&&... args)
64 {
65 static_cast<Handler&&>(handler_)(
66 static_cast<Arg&&>(arg),
67 static_cast<Args&&>(args)...);
68 }
69
70 template <typename... Args>
71 void operator()(const asio::error_code& ec, Args&&... args)
72 {
73 ec_ = ec;
74 static_cast<Handler&&>(handler_)(static_cast<Args&&>(args)...);
75 }
76
77//private:
78 asio::error_code& ec_;
79 Handler handler_;
80};
