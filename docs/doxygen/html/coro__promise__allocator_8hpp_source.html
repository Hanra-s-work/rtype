23template <typename AllocatorType>
24void* allocate_coroutine(const std::size_t size, AllocatorType alloc_)
25{
26 using alloc_type = typename std::allocator_traits<AllocatorType>::template
27 rebind_alloc<unsigned char>;
28 alloc_type alloc{alloc_};
29
30 const auto align_needed = size % alignof(alloc_type);
31 const auto align_offset = align_needed != 0
32 ? alignof(alloc_type) - align_needed : 0ull;
33 const auto alloc_size = size + sizeof(alloc_type) + align_offset;
34 const auto raw =
35 std::allocator_traits<alloc_type>::allocate(alloc, alloc_size);
36 new(raw + size + align_offset) alloc_type(std::move(alloc));
37
38 return raw;
39}

42template <typename AllocatorType>
43void deallocate_coroutine(void* raw_, const std::size_t size)
44{
45 using alloc_type = typename std::allocator_traits<AllocatorType>::template
46 rebind_alloc<unsigned char>;
47
48 const auto raw = static_cast<unsigned char *>(raw_);
49
50 const auto align_needed = size % alignof(alloc_type);
51 const auto align_offset = align_needed != 0
52 ? alignof(alloc_type) - align_needed : 0ull;
53 const auto alloc_size = size + sizeof(alloc_type) + align_offset;
54
55 auto alloc_p = reinterpret_cast<alloc_type *>(raw + size + align_offset);
56 auto alloc = std::move(*alloc_p);
57 alloc_p->~alloc_type();
58 std::allocator_traits<alloc_type>::deallocate(alloc, raw, alloc_size);
59}

67template <typename T, typename First, typename... Args>
68constexpr std::size_t variadic_first(std::size_t pos = 0u)
69{
