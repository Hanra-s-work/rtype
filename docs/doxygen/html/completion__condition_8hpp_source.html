29// The default maximum number of bytes to transfer in a single operation.
30enum default_max_transfer_size_t { default_max_transfer_size = 65536 };
31
32// Adapt result of old-style completion conditions (which had a bool result
33// where true indicated that the operation was complete).
34inline std::size_t adapt_completion_condition_result(bool result)
35{
36 return result ? 0 : default_max_transfer_size;
37}
38
39// Adapt result of current completion conditions (which have a size_t result
40// where 0 means the operation is complete, and otherwise the result is the
41// maximum number of bytes to transfer on the next underlying operation).
42inline std::size_t adapt_completion_condition_result(std::size_t result)
43{
44 return result;
45}

80class transfer_exactly_t
81{
82public:
83 typedef std::size_t result_type;
84
85 explicit transfer_exactly_t(std::size_t size)
86 : size_(size)
87 {
88 }
89
90 template <typename Error>
91 std::size_t operator()(const Error& err, std::size_t bytes_transferred)
92 {
93 return (!!err || bytes_transferred >= size_) ? 0 :
94 (size_ - bytes_transferred < default_max_transfer_size
95 ? size_ - bytes_transferred : std::size_t(default_max_transfer_size));
96 }
97
98private:
99 std::size_t size_;
