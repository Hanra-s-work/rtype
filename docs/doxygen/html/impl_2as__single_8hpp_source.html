32// Class to adapt a as_single_t as a completion handler.
33template <typename Handler>
34class as_single_handler
35{
36public:
37 typedef void result_type;
38
39 template <typename CompletionToken>
40 as_single_handler(as_single_t<CompletionToken> e)
41 : handler_(static_cast<CompletionToken&&>(e.token_))
42 {
43 }
44
45 template <typename RedirectedHandler>
46 as_single_handler(RedirectedHandler&& h)
47 : handler_(static_cast<RedirectedHandler&&>(h))
48 {
49 }
50
51 void operator()()
52 {
53 static_cast<Handler&&>(handler_)();
54 }
55
56 template <typename Arg>
57 void operator()(Arg&& arg)
58 {
59 static_cast<Handler&&>(handler_)(static_cast<Arg&&>(arg));
60 }
61
62 template <typename... Args>
63 void operator()(Args&&... args)
64 {
65 static_cast<Handler&&>(handler_)(
66 std::make_tuple(static_cast<Args&&>(args)...));
67 }
68
69//private:
70 Handler handler_;
71};
