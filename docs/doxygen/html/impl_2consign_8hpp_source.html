31// Class to adapt a consign_t as a completion handler.
32template <typename Handler, typename... Values>
33class consign_handler
34{
35public:
36 typedef void result_type;
37
38 template <typename H>
39 consign_handler(H&& handler, std::tuple<Values...> values)
40 : handler_(static_cast<H&&>(handler)),
41 values_(static_cast<std::tuple<Values...>&&>(values))
42 {
43 }
44
45 template <typename... Args>
46 void operator()(Args&&... args)
47 {
48 static_cast<Handler&&>(handler_)(static_cast<Args&&>(args)...);
49 }
50
51//private:
52 Handler handler_;
53 std::tuple<Values...> values_;
54};
