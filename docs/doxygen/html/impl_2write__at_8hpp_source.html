38 template <typename SyncRandomAccessWriteDevice, typename ConstBufferSequence,
39 typename ConstBufferIterator, typename CompletionCondition>
40 std::size_t write_at_buffer_sequence(SyncRandomAccessWriteDevice& d,
41 uint64_t offset, const ConstBufferSequence& buffers,
42 const ConstBufferIterator&, CompletionCondition completion_condition,
43 asio::error_code& ec)
44 {
45 ec = asio::error_code();
46 asio::detail::consuming_buffers<const_buffer,
47 ConstBufferSequence, ConstBufferIterator> tmp(buffers);
48 while (!tmp.empty())
49 {
50 if (std::size_t max_size = detail::adapt_completion_condition_result(
51 completion_condition(ec, tmp.total_consumed())))
52 {
53 tmp.consume(d.write_some_at(offset + tmp.total_consumed(),
54 tmp.prepare(max_size), ec));
55 }
56 else
57 break;
58 }
59 return tmp.total_consumed();
60 }

203 : base_from_cancellation_state<WriteHandler>(
204 static_cast<base_from_cancellation_state<WriteHandler>&&>(other)),
205 base_from_completion_cond<CompletionCondition>(
206 static_cast<base_from_completion_cond<CompletionCondition>&&>(other)),
207 device_(other.device_),

214
215 void operator()(asio::error_code ec,
216 std::size_t bytes_transferred, int start = 0)
217 {
218 std::size_t max_size;
219 switch (start_ = start)
220 {
221 case 1:
222 max_size = this->check_for_completion(ec, buffers_.total_consumed());
223 for (;;)
224 {
225 {
226 ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_write_at"));
227 device_.async_write_some_at(
228 offset_ + buffers_.total_consumed(), buffers_.prepare(max_size),
229 static_cast<write_at_op&&>(*this));
230 }
231 return; default:
232 buffers_.consume(bytes_transferred);
233 if ((!ec && bytes_transferred == 0) || buffers_.empty())
234 break;
235 max_size = this->check_for_completion(ec, buffers_.total_consumed());
236 if (max_size == 0)
237 break;
238 if (this->cancelled() != cancellation_type::none)
239 {
240 ec = asio::error::operation_aborted;
241 break;
242 }
243 }
244
245 static_cast<WriteHandler&&>(handler_)(
246 static_cast<const asio::error_code&>(ec),
247 static_cast<const std::size_t&>(buffers_.total_consumed()));
248 }
249 }
