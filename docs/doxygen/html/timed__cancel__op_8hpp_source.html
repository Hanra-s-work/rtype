41class timed_cancel_op

43public:

49 timed_cancel_op(Handler& handler, Timer timer,

52 handler_(static_cast<Handler&&>(handler)),
53 timer_(static_cast<Timer&&>(timer)),

59 }

63 if (has_payload_)

72 template <typename Initiation, typename... Args>
73 void start(Initiation&& initiation, Args&&... args)
74 {
75 using op_handler_type =
76 timed_cancel_op_handler<timed_cancel_op, Signatures...>;
77 op_handler_type op_handler(this);
78
79 using timer_handler_type =
80 timed_cancel_timer_handler<timed_cancel_op>;
81 timer_handler_type timer_handler(this);
82
83 associated_cancellation_slot_t<Handler> slot
84 = (get_associated_cancellation_slot)(handler_);
85 if (slot.is_connected())
86 cancel_proxy_ = &slot.template emplace<cancel_proxy>(this);
87
88 timer_.async_wait(static_cast<timer_handler_type&&>(timer_handler));
89 async_initiate<op_handler_type, Signatures...>(
90 static_cast<Initiation&&>(initiation),
91 static_cast<op_handler_type&>(op_handler),
92 static_cast<Args&&>(args)...);
93 }
94
95 template <typename Message>
96 void handle_op(Message&& message)
97 {
98 if (cancel_proxy_)
99 cancel_proxy_->op_ = nullptr;
100
101 new (&payload_storage_.payload_) payload_type(
102 static_cast<Message&&>(message));
103 has_payload_ = true;
104
105 if (has_pending_timer_wait_)
106 {
107 timer_.cancel();
108 release();
109 }
110 else
111 {
112 complete();
113 }
114 }
115
116 void handle_timer()
117 {
118 has_pending_timer_wait_ = false;
119
120 if (has_payload_)
121 {
122 complete();
123 }
124 else
125 {
126 cancellation_signal_.emit(cancellation_type_);
127 release();
128 }
129 }
130
131 void release()
132 {
133 if (--ref_count_ == 0)
134 {
135 ptr p = { asio::detail::addressof(handler_), this, this };
136 Handler handler(static_cast<Handler&&>(handler_));
137 p.h = asio::detail::addressof(handler);
138 p.reset();
139 }
140 }
141
142 void complete()
143 {
144 if (--ref_count_ == 0)
145 {
146 ptr p = { asio::detail::addressof(handler_), this, this };
147 completion_payload_handler<payload_type, Handler> handler(
148 static_cast<payload_type&&>(payload_storage_.payload_), handler_);
149 p.h = asio::detail::addressof(handler.handler());
150 p.reset();
151 handler();
152 }
153 }
