38 template <typename SyncWriteStream, typename ConstBufferSequence,
39 typename ConstBufferIterator, typename CompletionCondition>
40 std::size_t write(SyncWriteStream& s,
41 const ConstBufferSequence& buffers, const ConstBufferIterator&,
42 CompletionCondition completion_condition, asio::error_code& ec)
43 {
44 ec = asio::error_code();
45 asio::detail::consuming_buffers<const_buffer,
46 ConstBufferSequence, ConstBufferIterator> tmp(buffers);
47 while (!tmp.empty())
48 {
49 if (std::size_t max_size = detail::adapt_completion_condition_result(
50 completion_condition(ec, tmp.total_consumed())))
51 tmp.consume(s.write_some(tmp.prepare(max_size), ec));
52 else
53 break;
54 }
55 return tmp.total_consumed();
56 }

350 void operator()(asio::error_code ec,
351 std::size_t bytes_transferred, int start = 0)
352 {
353 std::size_t max_size;
354 switch (start_ = start)
355 {
356 case 1:
357 max_size = this->check_for_completion(ec, buffers_.total_consumed());
358 for (;;)
359 {
360 {
361 ASIO_HANDLER_LOCATION((__FILE__, __LINE__, "async_write"));
362 stream_.async_write_some(buffers_.prepare(max_size),
363 static_cast<write_op&&>(*this));
364 }
365 return; default:
366 buffers_.consume(bytes_transferred);
367 if ((!ec && bytes_transferred == 0) || buffers_.empty())
368 break;
369 max_size = this->check_for_completion(ec, buffers_.total_consumed());
370 if (max_size == 0)
371 break;
372 if (this->cancelled() != cancellation_type::none)
373 {
374 ec = error::operation_aborted;
375 break;
376 }
377 }
378
379 static_cast<WriteHandler&&>(handler_)(
380 static_cast<const asio::error_code&>(ec),
381 static_cast<const std::size_t&>(buffers_.total_consumed()));
382 }
383 }
