36template <typename Executor, typename Allocator>
37class executor::impl
38 : public executor::impl_base
39{
40public:
41 typedef ASIO_REBIND_ALLOC(Allocator, impl) allocator_type;
42
43 static impl_base* create(const Executor& e, Allocator a = Allocator())
44 {
45 raw_mem mem(a);
46 impl* p = new (mem.ptr_) impl(e, a);
47 mem.ptr_ = 0;
48 return p;
49 }
50
51 static impl_base* create(std::nothrow_t, const Executor& e) noexcept
52 {
53 return new (std::nothrow) impl(e, std::allocator<void>());
54 }
55
56 impl(const Executor& e, const Allocator& a) noexcept
57 : impl_base(false),
58 ref_count_(1),
59 executor_(e),
60 allocator_(a)
61 {
62 }
63
64 impl_base* clone() const noexcept
65 {
66 detail::ref_count_up(ref_count_);
67 return const_cast<impl_base*>(static_cast<const impl_base*>(this));
68 }
69
70 void destroy() noexcept

72 if (detail::ref_count_down(ref_count_))
