180 // Wait for the descriptor to become ready to read, ready to write, or to have
181 // pending error conditions.
182 asio::error_code wait(implementation_type& impl,
183 posix::descriptor_base::wait_type w, asio::error_code& ec)
184 {
185 switch (w)
186 {
187 case posix::descriptor_base::wait_read:
188 descriptor_ops::poll_read(impl.descriptor_, impl.state_, ec);
189 break;
190 case posix::descriptor_base::wait_write:
191 descriptor_ops::poll_write(impl.descriptor_, impl.state_, ec);
192 break;
193 case posix::descriptor_base::wait_error:
194 descriptor_ops::poll_error(impl.descriptor_, impl.state_, ec);
195 break;
196 default:
197 ec = asio::error::invalid_argument;
198 break;
199 }
200
201 ASIO_ERROR_LOCATION(ec);
202 return ec;
203 }

205 // Asynchronously wait for the descriptor to become ready to read, ready to
206 // write, or to have pending error conditions.
207 template <typename Handler, typename IoExecutor>
208 void async_wait(implementation_type& impl,
209 posix::descriptor_base::wait_type w,
210 Handler& handler, const IoExecutor& io_ex)
211 {
212 bool is_continuation =
213 asio_handler_cont_helpers::is_continuation(handler);
214
215 associated_cancellation_slot_t<Handler> slot
216 = asio::get_associated_cancellation_slot(handler);
217
218 // Allocate and construct an operation to wrap the handler.
219 typedef reactive_wait_op<Handler, IoExecutor> op;
220 typename op::ptr p = { asio::detail::addressof(handler),
221 op::ptr::allocate(handler), 0 };
222 p.p = new (p.v) op(success_ec_, handler, io_ex);
223
224 ASIO_HANDLER_CREATION((reactor_.context(), *p.p, "descriptor",
225 &impl, impl.descriptor_, "async_wait"));
226
227 int op_type;
228 switch (w)
229 {
230 case posix::descriptor_base::wait_read:
231 op_type = reactor::read_op;
232 break;
233 case posix::descriptor_base::wait_write:
234 op_type = reactor::write_op;
235 break;
236 case posix::descriptor_base::wait_error:
237 op_type = reactor::except_op;
238 break;
239 default:
240 p.p->ec_ = asio::error::invalid_argument;
241 start_op(impl, reactor::read_op, p.p,
242 is_continuation, false, true, false, &io_ex, 0);
243 p.v = p.p = 0;
244 return;
245 }
246
247 // Optionally register for per-operation cancellation.
248 if (slot.is_connected())
249 {
250 p.p->cancellation_key_ =
251 &slot.template emplace<reactor_op_cancellation>(
252 &reactor_, &impl.reactor_data_, impl.descriptor_, op_type);
253 }
254
255 start_op(impl, op_type, p.p, is_continuation,
256 false, false, false, &io_ex, 0);
257 p.v = p.p = 0;
258 }

260 // Write some data to the descriptor.
261 template <typename ConstBufferSequence>
262 size_t write_some(implementation_type& impl,
263 const ConstBufferSequence& buffers, asio::error_code& ec)
264 {
265 typedef buffer_sequence_adapter<asio::const_buffer,
266 ConstBufferSequence> bufs_type;
267
268 size_t n;
269 if (bufs_type::is_single_buffer)
270 {
271 n = descriptor_ops::sync_write1(impl.descriptor_,
272 impl.state_, bufs_type::first(buffers).data(),
273 bufs_type::first(buffers).size(), ec);
274 }
275 else
276 {
277 bufs_type bufs(buffers);
278
279 n = descriptor_ops::sync_write(impl.descriptor_, impl.state_,
280 bufs.buffers(), bufs.count(), bufs.all_empty(), ec);
281 }
282
283 ASIO_ERROR_LOCATION(ec);
284 return n;
285 }
